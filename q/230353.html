def k(c,p,V=0):
    while p[1:]!=[]:
        i=randrange(len(p))
        m,l,r=(p[i][0]/p[i][1],i),[],[]
        for j,t in enumerate(p):
            [r,l][(t[0]/t[1],j)>=m]+=[t]
        v,w=map(sum,zip(*l))
        c,V,p=[(c,V,l),(c-w,V+v,r)][w<=c]
    v,w=(p+[(0,1)])[0]
