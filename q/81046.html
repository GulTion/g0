using LinearAlgebra

function VM(n)
    return [ifelse(k == 1, 1, (j - 1)^(k - 1)) for j in 1:n, k in 1:n]
end

function f(n)
    V = nullspace(VM(n + 1) - I)
    V = V[:, 1] # Get the first column
    if V[end] < 0
        V = -V
    end
    return (V / maximum(V))[end:-1:1] # Normalize and reverse
end

for n in 0:10
    print(f(n), "\n")
end
