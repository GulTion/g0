object Main {
  def dayOfYear(y: Int, m: Int, d: Int): Int = {
    val monthOffsets = "03479cehkmpr".toList
    d + Integer.parseInt(monthOffsets(m - 1).toString, 36) + (m - 1) * 28 - (if ((y % (if (y % 25 == 0) 16 else 4) != 0) && m > 1) 1 else 0)
  }

  def main(args: Array[String]): Unit = {
    println(dayOfYear(2016, 07, 05))  // 187
    println(dayOfYear(2000, 03, 28))  // 88
    println(dayOfYear(0666, 06, 06))  // 157
    println(dayOfYear(6789, 10, 11))  // 284
    println(dayOfYear(0004, 04, 04))  // 95
    println(dayOfYear(1337, 07, 13))  // 194
  }
}
