a => {              // a[] = input array
  for(              // loop:
    x = 1;          //   start with x = 1
    a.some(v =>     //   stop when there's no value v in a[]
                    //   such that:
      ++q % v |     //     v is not a divisor of the current value q
                    //     (where q = x + i, i ≥ 1)
      v / q,        //     or v = q
      q = x++       //     starting with q = x (x incremented afterwards)
    );              //   end of some()
  );                // end of for()
  return a.map(_ => // return the final list
    x++             // x, x + 1, ...
  )                 //
}                   //
