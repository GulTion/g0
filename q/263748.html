import numpy as np

def my_conv(A):
 h,l = np.zeros((2,200000),np.float64)
 h[:100000] = A>>17
 l[:100000] = A&((1<<17)-1)
 L = np.fft.rfft(l)
 H = np.fft.rfft(h)
 m = np.fft.irfft(L*H)[:-1].astype(int).astype(object)
 l = np.fft.irfft(L*L)[:-1].astype(int).astype(object)
 h = np.fft.irfft(H*H)[:-1].astype(int).astype(object)
 return l+(m<<18)+(h<<34)

a = np.random.randint(0,10**10,10**5)
print(my_conv(a)[100000-1])
a=a.tolist()
print(sum(x*y for x,y in zip(a,a[::-1])))
