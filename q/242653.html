import java.time.LocalDate
import java.time.temporal.{TemporalAdjusters, ChronoUnit}

object Main {
  def main(args: Array[String]): Unit = {
    val input = List(
      "2021-01-31, 0      => 2021-01-31  (boring)",
      "2021-01-31, 1      => 2021-02-28  (no overflow)",
      "2020-12-31, 1      => 2021-01-31  (next year)",
      "2020-12-07, 1      => 2021-01-07  (next year)",
      "2021-01-31, -1     => 2020-12-31  (previous year)",
      "2021-01-30, -1     => 2020-12-30  (previous year)",
      "2021-01-01, -1     => 2020-12-01  (previous year)",
      "2020-12-30, 2      => 2021-02-28  (no overflow)",
      "2021-02-28, 1      => 2021-03-31  (end-of-month -> end-of-month)",
      "2021-09-30, 1      => 2021-10-31  (end-of-month -> end-of-month)",
      "2021-02-28, -12    => 2020-02-29  (end-of-month -> end-of-month)",
      "2020-02-28, 1      => 2020-03-28  (leap year - 28.02 is not end-of-month)",
      "1995-02-28, -1140  => 1900-02-28  (not a leap year)"
    )

    input.map { s =>
      val parts = s.split("=>")
      val leftSide = parts(0).trim
      val rightSide = parts(1).trim
      (leftSide.split(",").map(_.trim), rightSide.split("  ")(0).trim)
    }.foreach { case (line, expectedStr) =>
      val date = LocalDate.parse(line(0))
      val offset = line(1).toInt
      val expected = LocalDate.parse(expectedStr)
      val result = addMonthsAdjustEndOfMonth(date, offset)
      println(s"($date, $offset months) => $result, isCorrect: ${result == expected}")
    }
  }

  def isLastDayOfMonth(date: LocalDate): Boolean =
    date == date.withDayOfMonth(date.lengthOfMonth())

  def addMonthsAdjustEndOfMonth(date: LocalDate, months: Int): LocalDate = {
    val newDate = date.plusMonths(months)
    if (isLastDayOfMonth(date)) newDate.`with`(TemporalAdjusters.lastDayOfMonth())
    else newDate
  }
}
