int f(const int* arr, int size)
{
    int max = 0;
        
    for(int i=size; i-->0;)
    {
        const int x = arr[i];
        if (x > i)
        {
            const int r = x+i;
            max = (r > size) ? r : size;
        }
    }
        
    if (max >= size) 
        return 1;
    else 
        return 1 + f(arr + max, size - max);
}
