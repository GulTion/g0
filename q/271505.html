How does it work?

import collections
import itertools

def recover_order(prefixes):
    counters = [collections.Counter(p) for p in sorted(prefixes + [[]], key=len)]
    for c1, c2 in itertools.pairwise(counters):
        elem, = (c2 - c1).elements()
        yield elem

from collections import Counter as C
from itertools import pairwise as P

def f(p):
