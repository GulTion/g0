import collections
import itertools

def recover_order(prefixes):
    counters = [collections.Counter(p) for p in sorted(prefixes + [[]], key=len)]
    for c1, c2 in itertools.pairwise(counters):
        elem, = (c2 - c1).elements()
        yield elem
