Commented

d =>                       // d[] = dictionary (array of strings)
w =>                       // w[] = word (array of characters)
d.filter(W =>              // for each word W in d[]:
  w.some((c, i, [...a]) => //   for each character c at index i in w[],
                           //   with a[] = copy of w[]:
    a.splice(i, 1) +       //     remove the i-th character from a[]
                           //     (which is c), prepend it to
    a.sort()               //     a[] sorted in lexicographical order
    ==                     //     and test whether this is equal to
    c +                    //     c prepended to
    [...W].sort()          //     W[] sorted in lexicographical order
  )                        //   end of some()
)                          // end of filter()
