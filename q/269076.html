    # resulting in a binary string ('0' -> '111101111')
    # the binary string is converted into a decimal value
    # the decimal value is offset by 410 to reduce the length of the pixmap list

    # select the pixmap corresponding to the character
    # using the ascii value
    char_pixmap = pixmap[ord(input_char) - (48 if input_char < 'A' else 55)]
    
    # account for the offset
    char_pixmap += 410
    
    # build the binary string representing the pixmap for the characters
