function f(a, b, m, k, n) {
  let A = [ a, b ];

  for(let i = 2; i < n; i++) {
    A[i] = (A[i - 2] + A[i - 1]) % m;

    if(A[i] == b && A[i - 1] == a) {
      A = A.slice(0, -2);
      break;
    }
  }

  A = A.map((v, i) => [v, i]).sort((a, b) => a[0] - b[0]);

  let q = Math.floor(n / A.length),
      r = n % A.length,
      j = 0;

  do {
    k -= q + (A[j++][1] < r);
  } while(k >= 0);

  return A[j - 1][0];
}
