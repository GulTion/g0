def long_addition(a, b)
  carry = 0
  a.zip(b).reverse.map {|x, y|
    sum = x + y + carry
    carry = sum > 9 ? 1 : 0
    sum % 10
  }.push(carry).reverse
end

    floor=math.floor
    output=""
    carry=0
    for i=1,#A do
        L=#A-i+1
        sumResult=A:sub(L,L)+carry+B:sub(L,L)
        carry=sumResult/10
        output=floor(sumResult%10)..output
    end
    return floor(carry)..output
end
