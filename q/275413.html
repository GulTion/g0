import functools


@functools.lru_cache(maxsize=None)
def divisor_pairs(n):
    divisors = [d for d in range(1, n + 1) if n % d == 0]
    return list(zip(divisors, divisors[::-1]))


@functools.lru_cache(maxsize=None)
def tree_count(nodes, max_height):
    if nodes == 1:
        return int(max_height > 0)
    next_height = max_height - 1
    return sum(
        sum(tree_count(i, max_height) * m * tree_count(m, next_height) for d, m in divisor_pairs(nodes - i))
        for i in range(1, nodes)
    ) // (nodes - 1)
