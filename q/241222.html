object PariGpInScala {
  def main(args: Array[String]): Unit = {
    val inputList = List(1, 3, 4, 5, 6, 7, 8, 9, 10)

    inputList.foreach { n =>
      println(s"$n summands:")
      println(s"1 = ${f(n).map(x=>"1/"++x.toString)mkString(" + ")} \n")
    }
  }

  def f(n: Int): List[BigInt] = {
    (1 to n).map { i =>
      var a,q = BigInt("1")
      for (_ <- 1 to i) {
        a *= q
        q = 1 + a
      }
      q - (if (i == n) 1 else 0)
    }.toList
  }
}
