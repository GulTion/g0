fn find_pairs(r: i32, u: i32, d: i32) {
    let mut a = 0;
    let mut b = 0;
    
    loop {
        let z = a * u + b * d;
        
        if z > r {
            b += 1;
        } else if z == r {
            println!("{}", a - b);
            a += 1;
        } else {
            a += 1;
        }
    }
}
