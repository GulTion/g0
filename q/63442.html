100,            # push 0..99

  1.@           # push 1 twice, rotate counting var to the top

    {

        auto l = F<Z>::v;
        s << (l % 2 ? "" : "Fibo")
          << (l % 3 ? "" : "Nacci");
        (l % 2 && l % 3 ? s << l : s) << "\n";
    }
};

    S() { }
};
