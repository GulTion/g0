#include <bits/stdc++.h>
using V=std::vector<int>;
V f(V a) {
    int l=0,r=a.size(),n,j,i,m,b,L=r,s=sqrt(r);
    std::map<V,int> M;
    V v(L);
    for(int i=0;i+s<L;i++) {
        V x(a.begin()+i,a.begin()+i+s);
        v[i]=M[x]=M[x]?M[x]:~i;
    }
    while (l<r) {
        m = (l+r)/2;
        b=0;
        std::set<V> S;
        for (i=m;i<=L;i++) {
            V x;
            for(j=i-m;j<i;j+=j+s<i?s:1)x.push_back((j+s<i?v:a)[j]);
            S.insert(x).second||(n=i-m,b=1);
        }
        b?l=m+1:r=m;
    }
    return V(a.begin()+n,a.begin()+n-1+l);
}
